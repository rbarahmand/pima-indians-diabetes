os.makedirs(IMAGES_PATH, exist_ok=True)

def save_fig(fig_id, tight_layout=True, fig_extension="jpg", resolution=300):
    path = os.path.join(IMAGES_PATH, fig_id + "." + fig_extension)
    print("Saving figure", fig_id)
    if tight_layout:
        plt.tight_layout()
    plt.savefig(path, format=fig_extension, dpi=resolution)

from imblearn.pipeline import Pipeline
from imblearn.over_sampling import SMOTE
from sklearn.model_selection import cross_validate
from sklearn.model_selection import GridSearchCV

model = SGDClassifier(random_state=42, penalty="elasticnet", l1_ratio=0.1)
over = SMOTE()
steps = [("over", over), ("model", model)]
pipe = Pipeline(steps)
param_grid = {
    "over__k_neighbors": list(range(1, 10)),
    "model__alpha": list(np.arange(0.000005, 0.0001, 0.000005)),
}
grid = GridSearchCV(pipe, param_grid, cv=cv, scoring='roc_auc', return_train_score=True, n_jobs=-1, error_score='raise')
